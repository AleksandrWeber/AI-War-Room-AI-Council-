
describe('programmabilityvaultizability rollout integration', () => {
  it('reports programmabilityvaultizability capabilities and rollout readiness', async () => {
    const capabilities = await request(app.getHttpServer())
      .get('/api/programmabilityvaultizability/capabilities')
      .expect(200)

    expect(capabilities.body).toMatchObject({
      supportsProgrammabilityvaultizabilityRollout: true,
      supportsProgrammabilityvaultizabilityAdminTools: true,
      supportsBillingInvoiceProgrammabilityvaultizabilitySignals: true,
    })

    const rollout = await request(app.getHttpServer())
      .get('/api/programmabilityvaultizability/readiness')
      .expect(200)

    expect(rollout.body.status).toBe('ready')
  })

  it('returns programmabilityvaultizability admin summary for owners', async () => {
    const response = await request(app.getHttpServer())
      .get('/api/programmabilityvaultizability/workspace/workspace_1/admin')
      .set(authHeaders)
      .expect(200)

    expect(response.body).toMatchObject({
      workspaceId: 'workspace_1',
      role: 'owner',
      stats: {
        totalDomains: 4,
        coveredDomains: expect.any(Number),
        programmabilityvaultizabilityPercent: expect.any(Number),
      },
    })
  })

  it('rejects programmabilityvaultizability admin tools for members', async () => {
    await request(app.getHttpServer())
      .get('/api/programmabilityvaultizability/workspace/workspace_1/admin')
      .set({
        'x-user-id': 'user_member',
        'x-workspace-id': 'workspace_1',
      })
      .expect(403)
  })
})
